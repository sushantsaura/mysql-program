DELIMITER $$
CREATE PROCEDURE EMP_RENT()
BEGIN
DECLARE CURR_SAL DECIMAL(9,2);
DECLARE V_emp_name VARCHAR(50);
DECLARE V_SALARY INT;
DECLARE V_FINISHED INTEGER DEFAULT 0;
DECLARE C3 CURSOR FOR SELECT emp_id,salary FROM employee;
DECLARE CONTINUE HANDLER FOR NOT FOUND SET V_FINISHED=2;
OPEN C3;
EMP: LOOP
FETCH C3 INTO V_emp_name,V_salary;
SET CURR_SAL=V_salary-(V_salary*0.1);
SELECT CONCAT(V_emp_name,CURR_SAL);
IF V_FINISHED=2 THEN 
LEAVE EMP;
END IF;
END LOOP EMP;
CLOSE C3;
END $$
DELIMITER $$
